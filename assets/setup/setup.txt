Node
curl -sL https://rpm.nodesource.com/setup_12.x | sudo bash -
yum -y install nodejs; node -v

Redis
wget http://download.redis.io/releases/redis-5.0.5.tar.gz
tar xvzf redis-5.0.5.tar.gz; cd redis-5.0.5; make; make install
run utils/install_server.sh
Edit /etc/redis/6379.conf and change "bind 127.0.0.1" to "bind 192.168.0.232"
service redis_6379 restart

Firewall
On the application servers
firewall-cmd --permanent --zone=public --add-port=8080/tcp

On the load balancer server
firewall-cmd --permanent --zone=public --add-port=80/tcp
firewall-cmd --permanent --zone=public --add-port=443/tcp

On the database servers
firewall-cmd --permanent --zone=public --add-port=1521/tcp
firewall-cmd --permanent --zone=public --add-port=6379/tcp

On all of the servers
firewall-cmd --reload
